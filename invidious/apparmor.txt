#include <tunables/global>

profile invidious flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for Invidious and PostgreSQL
  /invidious/invidious cx -> invidious_service,
  /usr/bin/postgres cx -> postgres_service,

  profile invidious_service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_invidious,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /share/** rw,

    # Access required for Invidious functionality
    /invidious/** r,
    /invidious/invidious rix,
    /invidious/config/** rw,
    /bin/bash rix,
    /etc/passwd r,
    /etc/group r,
    /dev/tty rw,
    
    # Network access for web interface
    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
  }
  
  profile postgres_service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>
    
    # PostgreSQL access
    /usr/bin/postgres rix,
    /usr/lib/postgresql/** r,
    /data/postgresql/** rwk,
    /var/lib/postgresql/** rwk,
    /tmp/** rwk,
    
    # Network access
    network inet stream,
    network inet6 stream,
  }
}
